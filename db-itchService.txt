create database servicios_itch
use servicios_itch

create table edificio (
idEdificio int identity(1, 1) primary key,
nomenclatura varchar(2),
ubicacion varchar(100),
)

create table departamentos (
idDepartamento int identity(1, 1) primary key,
departamento varchar(100),
idEdificio int references edificio (idEdificio),
)

create table areas (
idArea int identity(1, 1) primary key,
area varchar(100),
idDepartamento int references departamentos (idDepartamento),
)

create table perfilUsuarios (
idPerfilUsuario int identity(1, 1) primary key,
perfilUsuario varchar(30),
descripcion varchar(200),
)

create table escolaridad (
idEscolaridad int identity(1, 1) primary key,
escolaridad varchar(30),
)

create table perfilEmpleados (
idPerfilEmpleado int identity(1, 1) primary key,
perfilEmpleado varchar(50),
descripcion varchar(200),
)


create table usuarios(
idUsuario int identity(1,1) primary key,
idPerfilUsuario int references perfilUsuarios (idPerfilUsuario),
primerNombre varchar(20),
segundoNombre varchar(20),
primerApellido varchar(20),
segundoApellido varchar(20),
edad varchar(2),
idEscolaridad int references escolaridad (idEscolaridad),
telefono varchar(10),
celular varchar(10),
email varchar(50),
)

create table empleados(
idEmpleado int identity(1,1) primary key,
primerNombre varchar(20),
segundoNombre varchar(20),
primerApellido varchar(20),
segundoApellido varchar(20),
edad varchar(2),
idEscolaridad int references escolaridad (idEscolaridad),
telefono varchar(10),
celular varchar(10),
email varchar(50),
idPerfilEmpleado int references perfilEmpleados (idPerfilEmpleado),
idArea int references areas (idArea),
)

create table periodos(
idPeriodo int identity (1,1) primary key,
periodo varchar (20),
descripcionPeriodo varchar (100),
)

create table servicios(
idServicio int identity(1,1) primary key,
servicio varchar(50),
descripcionServicio varchar(500),
requisitosSolicitados varchar(500),
idArea int references areas (idArea),
idempleado int references empleados (idempleado),
idPeriodo int references periodos (idPeriodo), 
fecha varchar(50),
horario varchar(50),
)
